// Function to toggle the contact options visibility function toggleContactOptions() { var options = document.getElementById("contact-options"); // Toggle the display of the contact options if (options.style.display === "block") { options.style.display = "none"; } else { options.style.display = "block"; } } // Close contact options when clicking outside of them document.addEventListener('click', function(event) { var contactOptions = document.getElementById('contact-options'); var contactButton = document.querySelector('.buttons a'); // If the clicked element is not the contact button or inside the contact options box if (!contactButton.contains(event.target) && !contactOptions.contains(event.target)) { contactOptions.style.display = 'none'; } });
